[docs]    def setUp(self):
        nest.set_verbosity('M_WARNING')
        nest.ResetKernel()

        # setup basic circuit
        self.parrot = nest.Create("parrot_neuron", 2)

        # setup two generators converging onto the first parrot, i.e. multiple spikes
        self.generateSpikes([self.parrot[0]], [2.])
        self.generateSpikes([self.parrot[0]], [2.])

        # setup one generator converging onto the second parrot, i.e single spike
        self.generateSpikes([self.parrot[1]], [2.])




[docs]    def generateSpikes(self, neuron, times):
        """Trigger spike to given neuron at specified times."""
        delay = 1.
        gen = nest.Create("spike_generator", 1, {"spike_times": [t-delay for t in times]})
        nest.Connect(gen, neuron, syn_spec={"delay": delay})

[docs]def add_test(model_str):
    def test_func(self):
        m = nest.Create(model_str, 2)

        try:
            nest.Connect(self.parrot, m, 'one_to_one')
        except nest.NESTError as e:
            if "IllegalConnection" in e.message:
                # not a node we can connect to
                pass
            elif "IncompatibleReceptorType" in e.message:
                print "IncompatibleReceptorType"

        nest.Simulate(3.)

    test_func.__doc__ = 'Check multiplicity for \"%s\"' % model_str
    tmp = 'test_multiplicity_%s' % model_str
    test_func.__name__ = tmp
    setattr(NodeMultiplicityTestCase, test_func.__name__, test_func)

    models = nest.Models(mtype='nodes')
